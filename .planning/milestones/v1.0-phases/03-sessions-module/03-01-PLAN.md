**Private helper: `buildQueryString(params)`**
```typescript
private buildQueryString(params: SessionListParams): string {
  const qs = new URLSearchParams();
  if (params.limit !== undefined) qs.set('limit', String(params.limit));
  if (params.offset !== undefined) qs.set('offset', String(params.offset));
  if (params.status !== undefined) qs.set('status', params.status);
  const str = qs.toString();
  return str ? `?${str}` : '';
}
```

**Private helper: `wrapPaginated(raw, params)`**
```typescript
private wrapPaginated(
  raw: Session[] | PaginatedResponse<Session>,
  params: SessionListParams,
): PaginatedResponse<Session> {
  if (Array.isArray(raw)) {
    return {
      data: raw,
      limit: params.limit ?? raw.length,
      offset: params.offset ?? 0,
    };
  }
  return raw;
}
```
